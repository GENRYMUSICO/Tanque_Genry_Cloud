def confirmar_salida_premium():
    """Ventana de advertencia estilo Búnker con el escudo de seguridad"""
    import tkinter as tk
    from tkinter import messagebox
    import sys

    root = tk.Tk()
    root.withdraw() 
    
    respuesta = messagebox.askyesno(
        "AVISO IMPORTANTE - GRUPO GENRY", 
        "¿Desea desactivar la protección del Antivirus Genry?\n\nSu PC quedará Vulnerable."
    )
    
    if respuesta:
        sys.exit()
    else:
        root.destroy()

def iniciar_tray():
    import pystray
    from PIL import Image
    import webbrowser
    
    try:
        # Cargamos el logo oficial
        image = Image.open("LOGO AV GENRY OFICIAL.png")
        
        # Enlaces de soporte y radio
        URL_RADIO = "https://www.youtube.com/@genrymusico"
        URL_PAGO_1USD = "https://www.paypal.com" 
        URL_PAGO_8USD = "https://www.paypal.com"

        menu = pystray.Menu(
            # 1. Opción musical
            pystray.MenuItem("♬ ESCUCHAR: GRUPO GENRY", lambda: webbrowser.open(URL_RADIO)),
            pystray.Menu.SEPARATOR,
            
            # 2. Estado del Sistema (Fila corregida: Rojo y sin la fila de Usuario)
            pystray.MenuItem("Sistema Protegido", lambda: None, enabled=False), 
            
            pystray.Menu.SEPARATOR,
            # 3. Financiamiento
            pystray.MenuItem("💳 PLAN $1 USD (Mensual)", lambda: webbrowser.open(URL_PAGO_1USD)),
            pystray.MenuItem("🏆 PLAN $8 USD (Anual - PRO)", lambda: webbrowser.open(URL_PAGO_8USD)),
            
            pystray.Menu.SEPARATOR,
            # 4. Salida
            pystray.MenuItem("❌ SALIR DEL TANQUE", confirmar_salida_premium)
        )
        
        icon = pystray.Icon("AV_Genry", image, f"AV GENRY - Sistema Definitivo", menu)
        icon.run()
        
    except Exception as e:
        print(f"Error en el Tanque: {e}")
