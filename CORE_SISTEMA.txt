# --- 1. MÓDULOS DE VOZ Y SEGURIDAD ---
def hablar(texto):
    """Módulo de voz oficial con velocidad 165"""
    try:
        engine = pyttsx3.init()
        engine.setProperty('rate', 165)
        engine.say(texto)
        engine.runAndWait()
    except: pass

def calcular_hash(ruta):
    """Motor de detección de sapazos[cite: 6]"""
    h = hashlib.md5()
    try:
        with open(ruta, 'rb') as f:
            for chunk in iter(lambda: f.read(4096), b""): h.update(chunk)
        return h.hexdigest()
    except: return None

# --- 2. MOTOR DE PATRULLAJE ---
class MotorGenryCentinela:
    """Clase rescatada para patrullar en segundo plano[cite: 5]"""
    def __init__(self):
        self.lista_negra = set()
        if os.path.exists("hashes_virus.txt"):
            with open("hashes_virus.txt", "r") as f:
                self.lista_negra = {l.strip() for l in f if l.strip()}

    def patrullar(self):
        while True:
            for p in psutil.process_iter(['name', 'exe']):
                try:
                    if p.info['exe'] and calcular_hash(p.info['exe']) in self.lista_negra:
                        p.terminate()
                except: continue
            time.sleep(5)
