def hablar(texto):
    """Módulo de voz con velocidad optimizada"""
    try:
        engine = pyttsx3.init()
        engine.setProperty('rate', 165)
        engine.say(texto)
        engine.runAndWait()
    except: pass

def calcular_hash(ruta):
    """Motor de detección de sapazos"""
    h = hashlib.md5()
    try:
        with open(ruta, 'rb') as f:
            for chunk in iter(lambda: f.read(4096), b""): h.update(chunk)
        return h.hexdigest()
    except: return None
