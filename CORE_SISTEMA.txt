        pystray.Menu.SEPARATOR,

        pystray.MenuItem("🏆 PLAN PREMIUM $20", lambda: webbrowser.open("https://wise.com/pay/r/iLwaYmSjRQKpqb4")),
        pystray.Menu.SEPARATOR,
        pystray.MenuItem("❌ APAGAR BÚNKER", lambda icon, item: (hablar("Desactivando."), icon.stop(), os._exit(0)))
    )
    # Intenta cargar el logo oficial que bajamos de la nube
    try:
        img = Image.open(RUTA_LOGO_OFICIAL).convert("RGBA").resize((32, 32), Image.LANCZOS)
