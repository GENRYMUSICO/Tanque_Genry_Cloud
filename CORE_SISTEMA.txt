def iniciar_tray():
    """Crea el icono en la barra de tareas"""
    try:
        image = Image.open("LOGO AV GENRY OFICIAL.png")
    except:
        image = Image.new('RGB', (64, 64), color=(0, 255, 0)) # Cuadrado verde si no hay logo

    menu = pystray.Menu(
        pystray.MenuItem("Búnker Protegido", lambda: None, enabled=False),
        pystray.MenuItem("Salir del Tanque", confirmar_salida_premium)
    )
    icon = pystray.Icon("AV_Genry", image, f"Sistema {FONETICA_NOMBRE}", menu)
    icon.run()
